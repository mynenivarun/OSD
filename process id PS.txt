void getprocdata()
{
int i = 0;
struct uproc up;
printf(STDOUT, 
"Name\tpid\tppid\tsize\t\twaiting_on_channel\tkilled\n"); 
printf(STDOUT, "--------------------------------------------------------------------------
---\n");
for (i=0; i<NPROC; i++) {

}

int
sys_getprocinfo()
{
int proc_num, size = sizeof(struct uproc);
struct uproc *up;
if ((argint(0, &proc_num) < 0) || (argptr(1, (char **)&up, size) < 0))
return -1;
return getprocinfo(proc_num, up);
}

int
getprocinfo(int proc_num, struct uproc *up)
{
struct proc *p;
if (proc_num >= NPROC)
return -1;
p = &ptable.proc[proc_num];
memset(up, 0, sizeof(struct uproc))
memmove(up->name, p->name, 16);
up->pid = p->pid;
up->state = p->state;
if ((up->state != UNUSED)) {
up->ppid = p->parent->pid;
up->sz = p->sz;
}
if (up->pid == 1) {
up->sz = p->sz;
up->ppid = 0;
if (p->chan)
up->wait = 1;
else
up->wait = 0;
up->killed = p->killed;
return 0;
}
